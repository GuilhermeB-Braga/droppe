model Session {
  id        String   @id @default(auto()) @map("_id") @db.ObjectId
  name      String
  code      String   @unique
  createdAt DateTime @default(now())
  files     File[]
}

model File {
  id           String   @id @default(auto()) @map("_id") @db.ObjectId
  originalName String
  savedName    String
  fileSize     Int
  path         String
  sessionId    String   @db.ObjectId
  createdAt    DateTime @default(now())
  session      Session  @relation(fields: [sessionId], references: [id], onDelete: Cascade)
}
